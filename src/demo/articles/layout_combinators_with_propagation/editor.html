ob = (v) => {
    const listeners = [];
    return {
        on: (l) => listeners.push(l),
        get: () => v,
        set: (newV) => {
            // if my value didn't change, stop propagating
            if (toStr(v)===toStr(newV)) return;
            v = newV;
            listeners.forEach((l) => l(v));
        },
    };
};
to = (a, b, f) => {
    // when the value of a changes, set the value of b=f(a.get())
    a.on((v) => b.set(f(v)));
    // optional: set b=f(a.get()) right away
    b.set(f(a.get()));
};
and = (...obs) => {
    const res = ob(Array(obs.length))
    obs.forEach((o,i)=>to(o,res,v=>res.get().with(i,v)))
    return res
} 
eq = (a, b) => {
    to(a,b,v=>v)
    to(b,a,v=>v)
}
set = (b,a) => to(a,b,v=>v)
lift = (f) => (...obs) => {
    const res = ob()
    to(and(...obs),res,(args)=>f(...args))
    return res
}

above = (a, b) => {
    const w = max(a.w,b.w)
    const h = add(a.h,b.h)

    const x = ob(0)
    const y = ob(0)
    eq(x,a.x)
    eq(x,b.x)

    eq(y,a.y)
    set(b.y, add(y,a.h))
    set(y, sub(b.y,a.h))

    return {
        x, y, w, h,
        draw: (ctx) => {
            a.draw(ctx);
            b.draw(ctx);
        },
    };
};
